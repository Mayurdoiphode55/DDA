#include<iostream>
using namespace std;

int fib_rec(int n)
{
    if(n<=1) return n;
    return fib_rec(n-1) + fib_rec(n-2);
}

int fib_it(int n)
{
    if(n<=1)return n;
    
    int a=0, b=1, c=a+b;
    for(int i=2; i<n; i++)
    {
        c = a+b;
        a=b;
        b=c;
    }
    return b;
}

int main()
{
    int n;
    cout<<"enter the number"<<endl;
    cin>>n;
    
    int choice;
    cout<<"enter the choice \n1. recursive: \n2. iterative: "<<endl;
    cin>>choice;
    
    cout<<"fibonacci series"<<endl;
    
    if(choice == 1)
    {
        for(int i=0; i<n; i++)
        {
            cout<<fib_rec(i)<<" ";
        }
    }
    else if(choice == 2)
    {
        for(int i=0; i<n; i++)
        {
            cout<<fib_it(i)<<" ";
        }
    }
    else{
        cout<<"invalid choice";
    }
    cout<<endl;
    return 0;
    
}
